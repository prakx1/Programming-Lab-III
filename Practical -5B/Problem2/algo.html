function myfun() {
var input = document.getElementById("arr").value;
if (input) {
var array = input.split(",");
var max = parseInt(array[0]);
var min = parseInt(array[0]);
for (var i = 0; i < array.length; i++) {
current = parseInt(array[i]);
if (current > max) {
max = current;
}
if (current < min) {
min = current;
}
}
document.getElementById(
"inputArray"
).innerHTML = `You have entered:${array}`;
document.getElementById(
"result"
).innerHTML = `Min=${min} and Max=${max}`;
console.log(array);
}
}
